SELECT `a`.`id`, 
`a`.`application_id`, 
`a`.`catalog_type_id`, 
`a`.`source_id`, 
`a`.`enabled`, 
`a`.`redirect_to_id`, 
`a`.`sef_request`, 
`a`.`page_type`, 
`a`.`extension_instance_id`, 
`a`.`view_group_id`, 
`a`.`primary_category_id`
FROM `molajo_catalog` AS `a`
WHERE `a`.`enabled` = '1'
AND `a`.`redirect_to_id` = ''
